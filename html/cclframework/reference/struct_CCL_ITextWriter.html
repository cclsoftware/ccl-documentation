struct CCL::ITextWriter

Overview

#include <itextwriter.h>

struct ITextWriter: public CCL::IUnknown
{
    // methods

    virtual void CCL_API setDocumentLineFormat (TextLineFormat lineFormat) = 0;

    virtual tresult CCL_API beginDocument (
        IStream& stream,
        TextEncoding encoding = Text::kUnknownEncoding
    ) = 0;

    virtual tresult CCL_API endDocument () = 0;
    virtual tresult CCL_API writeLine (StringRef text) = 0;

Detailed Documentation

Methods

virtual void CCL_API setDocumentLineFormat (TextLineFormat lineFormat) = 0

Set line ending style.

Needs to be called before beginDocument().

virtual tresult CCL_API beginDocument (
    IStream& stream,
    TextEncoding encoding = Text::kUnknownEncoding
) = 0

Begin new document.

Supported encodings: “UTF-16” (default), “UTF-8”, “ISO-8859-1”, “US-ASCII”.

virtual tresult CCL_API endDocument () = 0

Finish writing to stream (see beginDocument)

virtual tresult CCL_API writeLine (StringRef text) = 0

Write line of text.
