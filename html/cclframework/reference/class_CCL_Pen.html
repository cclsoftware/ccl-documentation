Pen definition.

#include <pen.h>

class Pen: protected CCL::PlainPen
{
public:
    // typedefs

    typedef float Size;

    // enums

    enum
    {
        kPenTypeMask  = 0xff,
        kLineCapMask  = 0xff < <8,
        kLineJoinMask = 0xff < <16,
    };

    enum LineCap;
    enum LineJoin;
    enum PenType;

    // fields

     color;

    // construction

    Pen (Color _color = Colors::kBlack, Size _width = Size (1));

    // methods

    PROPERTY_BY_VALUE (int32, style, Style);
    Color PROPERTY_BY_VALUE (Size, width, Width);
    void setLineCap (LineCap cap);
    void setLineJoin (LineJoin join);
    PenType getPenType () const;
    LineCap getLineCap () const;
    LineJoin getLineJoin () const;
    bool isEqual (PenRef pen) const;
    bool operator == (PenRef other) const;
    bool operator != (PenRef other) const;
};
