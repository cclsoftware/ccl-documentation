struct TouchEvent: public CCL::PointerEvent
{
    // enums

    enum
    {
        kNoTouchId = 0,
    };

    enum EventType;

    // fields

    ITouchCollection& touches;
    TouchID touchID;

    // construction

    TouchEvent (
        ITouchCollection& touches,
        int eventType = kBegin,
        const KeyState& keys = KeyState (),
        InputDevice inputDevice = kPointerInput
    );

    // methods

    bool isHoverEvent () const;
};
