enum CCL::ITypeInfo::DataTypes

Overview

Data types. More…

#include <iobject.h>

enum DataTypes
{
    kVoid      = 0,
    kPrimitive = 0x01,
    kObject    = 0x02,
    kContainer = 0x03,
    kComposite = 0x04,
    kVariant   = 0x05,
    kInt       = kPrimitive | 0x0100,
    kFloat     = kPrimitive | 0x0200,
    kString    = kPrimitive | 0x0300,
    kBool      = kPrimitive | 0x0400,
    kBlob      = kPrimitive | 0x0500,
    kReadOnly  = 0x010000,
};
