enum Core::Errors::ErrorCodes

Overview

#include <coreproperty.h>

enum ErrorCodes
{
    kError_NoError         = 0,
    kError_ItemNotFound    = 100,
    kError_InvalidArgument,
    kError_InvalidThread,
    kError_OutOfMemory,
    kError_InvalidState,
    kError_NotReady,
    kError_NotImplemented,
    kError_Failed,
};
