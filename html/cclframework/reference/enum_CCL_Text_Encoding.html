enum CCL::Text::Encoding

Overview

Text Encodings. More…

#include <textencoding.h>

enum Encoding
{
    kUnknownEncoding = 0,
    kASCII           = 'ASCI',
    kISOLatin1       = 'Lat1',
    kWindowsLatin1   = 'ANSI',
    kDOSLatinUS      = 'C437',
    kMacRoman        = 'MacR',
    kShiftJIS        = 'SJIS',
    kUTF8            = 'UTF8',
    kUTF16LE         = 'U16L',
    kUTF16BE         = 'U16B',
    kUTF16           = kUTF16LE,
    kSystemEncoding  = 'Syst',
};
