    bool isInt () const;

    Variant& operator = (tresult t);
    Variant& operator = (int64 l);
    Variant& operator = (uint64 l);
    Variant& operator = (int l);

    Variant& operator = (int16 l);
    Variant& operator = (uint16 l);
    Variant& operator = (int8 l);
    Variant& operator = (uint8 l);
