enum CCL::IFileIterator::Modes

Overview

#include <ifilesystem.h>

enum Modes
{
    kFiles          = 1 < <0,
    kFolders        = 1 < <1,
    kAll            = kFiles|kFolders,
    kBundlesAsFiles = 1 < <2,
    kIgnoreHidden   = 1 < <3,
};

Detailed Documentation

Enum Values

kFiles

iterate files

kFolders

iterate folders

kAll

iterate files + folders

kBundlesAsFiles

do not descend into bundles

kIgnoreHidden

ignore hidden files/folders
